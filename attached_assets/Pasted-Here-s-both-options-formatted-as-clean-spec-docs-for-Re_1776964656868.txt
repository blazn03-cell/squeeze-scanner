Entry Point — index.js
javascriptimport { runScan } from './scanner.js';
import { logToSheet } from './sheets.js';
import { WATCHLIST } from './config.js';

async function main() {
  const timestamp = new Date().toISOString();
  console.log(`[${timestamp}] Scan starting for ${WATCHLIST.length} tickers`);
  
  try {
    const results = await runScan(WATCHLIST);
    await logToSheet(timestamp, results);
    console.log(`[${timestamp}] Logged ${results.length} candidates`);
  } catch (err) {
    console.error(`[${timestamp}] SCAN FAILED:`, err.message);
    await logToSheet(timestamp, [{ 
      ticker: 'ERROR', 
      score: 0, 
      thesis: err.message 
    }]).catch(() => {});
    process.exit(1);
  }
}

main();
